package citeunseen;
 
import java.util.concurrent.TimeUnit;
 
public class Timer {
	private String name;
 	private long startTime;
	private long endTime;
	private long totalTime;

	// Static methods
	//
	private static double s (long time) {
		return ms(time) / 1000.0;
	}
	
	private static double ms (long time) {
		return TimeUnit.MILLISECONDS.convert(time, TimeUnit.NANOSECONDS);
	}	

	public static Timer startNew () {
		return new Timer();
	}
	public static Timer startNew (String name) {
		return new Timer(name);
	}
	
	// Constructors
	public Timer () {
		this.start();
	}
	
	public Timer (String name) {
		this.name = name;
		start();
	}
	
 	// Non-static methods for running a timer
	//
	public void start () {
		startTime = System.nanoTime();
		if (name != null) Dev.out.println(name+" started...");		
	}
	
	public double stop () {
		endTime = System.nanoTime();		
		long runTime = endTime - startTime;
		totalTime += runTime;
	
		if (name != null) Dev.out.println(name+" run time: "+ms(runTime)+"s ");
		
		return ms(runTime);
	}
	
	public double total () {
		return ms(totalTime);
	}
}
